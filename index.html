Enhanced Usage Example:

// Fetch enhanced compliance insights
async function getEnhancedComplianceStatus() {
    const base = 'https://meridian-knowledge-solutions.github.io/fedramp_20x_public_submission_live';
    
    // Traditional compliance data
    const validations = await fetch(`${base}/unified_ksi_validations.json`);
    const ksiData = await validations.json();
    
    // Enhanced POA&M insights
    const poamData = await fetch(`${base}/enhanced_poam_data.json`);
    const threeTierPOAM = await poamData.json();
    
    return {
        compliance_score: calculateComplianceScore(ksiData),
        enhancement_opportunities: threeTierPOAM.enhancement_tracking.length,
        staleness_items: threeTierPOAM.staleness_tracking.length,
        total_improvements: threeTierPOAM.total_proactive_items,
        innovation_score: calculateInnovationScore(threeTierPOAM)
    };
}
